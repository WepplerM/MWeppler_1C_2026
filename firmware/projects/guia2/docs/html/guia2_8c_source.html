100void MostrarDisplay(uint8_t *bcdArray){
101
102 for(int i = 0; i < DIGITS; i++) {
103
104 // Apagar todos los dígitos
105 for(int j = 0; j < DIGITS; j++){
106 GPIOState(selGpio[j], DIGIT_OFF);
107 }
108
109 // Enviar BCD
110 setBcdToGpio(bcdArray[i], bcdGpio);
111
112 // Activar dígito
113 GPIOState(selGpio[i], DIGIT_ON);
114
115 // Retardo de multiplexado
116 for(volatile int k = 0; k < MUX_DELAY; k++);
117 }
118}

120void Task_Medicion(void *pvParameter){
121
122 while(true){
123
124 if (medir){
125
126 distancia = HcSr04ReadDistanceInCentimeters();
127
128 if (!mantener){
129 distancia_hold = distancia;
130 }
131
132 // LEDs con valor estable
133 if (distancia_hold < DIST_MIN){
134 LedsOffAll();
135 }
136 else if (distancia_hold < DIST_MED){
137 LedsOffAll();
138 LedOn(LED_1);
139 }
140 else if (distancia_hold < DIST_MAX){
141 LedsOffAll();
142 LedOn(LED_1);
143 LedOn(LED_2);
144 }
145 else{
146 LedsOffAll();
147 LedOn(LED_1);
148 LedOn(LED_2);
149 LedOn(LED_3);
150 }
151 }
152 else{
153 LedsOffAll();
154 }
155
156 vTaskDelay(1000 / portTICK_PERIOD_MS); // 🔥 EXACTO 1 segundo
157 }

180void Task_Botones(void *pvParameter){

182 bool estado_ant_1 = false;

189
190 // FLANCO botón medir

192 medir = !medir;
193 }
194
195 // FLANCO botón mantener
196 if (estado_2 && !estado_ant_2){
197 mantener = !mantener;
198 }
199
200 estado_ant_1 = estado_1;
201 estado_ant_2 = estado_2;
202
203 vTaskDelay(50 / portTICK_PERIOD_MS);
204 }
